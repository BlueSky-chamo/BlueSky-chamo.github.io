#前言#
学习tcl时作一些练习，方便查找

#求和#

# function :区间求和
# para1:区间左边界，即最小值
# para2:区间右边界，即最大值，缺省为100
# return:区间所有数相加之和
proc Get_Sum { min {max 100}} {
    set sum 0;
    for {set i $min} {$i<=$max} {incr i} {
        incr sum $i
    }
    return $sum
}

puts [Get_Sum 1]

#求平均值#
对于任意区间的平均值，无法完成浮点运算，因此只能取模

# function :求平均值
# para1:区间左边界，即最小值
# para2:区间右边界，即最大值，缺省为100
# return:区间所有数的平均值
proc Get_Average { min {max 10}} {
    set sum 0;
    for {set i $min} {$i<=$max} {incr i} {
        incr sum $i
    }
    set average [expr $sum/[expr $max-$min+1]]
    return $average
}

puts [Get_Average 1]

任意多个整数的平均值：

proc Get_Average {args} {
    set i 0;
    set sum 00;
    foreach n $args {
        incr i 1;
        incr sum $n;
    }
    set avg [expr $sum/$i]
    puts "$sum,$avg"
}

Get_Average 2 4 6 8 10

#求最大值#
要求输入的参数的格式为 小明 22 小李 88 张华 99 返回其中年龄最大的姓名

proc Get_Max {args} {
    set i 0
    set name ""
#{n a}与输入参数匹配，小明 22代表第一个参数
    foreach {n a} $args {
        if {$a>$i} {
            set i $a
            set name $n
        }
    }
    return "$name $a"

}

puts [Get_Max 小明 22 小李 88 张华 99]

随机数

1.生成某区间的随机数

proc Random_Range {min max} {
    set random [expr round(($max-$min)*rand()+$min)];#round为四舍五入最接近的整数值
    return $random
}

puts [Random 1 10]

2.生成num个min到max之间的不重复的随机整数
proc Random_Range {min max} {
set random [expr round(($max-$min)*rand()+$min)]
return $random
}

proc Random_Unique {min max num} {
    array set arr {};#空数组
    #个数不能大于区间数个数
    if {$num>[expr $max-$min+1]} {
        return ""
    }
    #数组大小小于随机数个数
    while {[array size arr] <$num} {
        set x [Random_Range $min $max]
        puts $x
        set arr($x) ""

    }

    return [array name arr]
}

puts [Random_Unique 1 10 3]

proc Random_List {llist} {
    set id [Random_Range 0 [expr [llength $llist]-1]]
    return [lindex $llist $id]
}

进制转换

二进制转换十进制

proc Binary_Decimal {Binary} {
    set sum 0
    set len [string length $Binary]
    for {set i 0} {$i<$len} {incr i} {
        incr sum [expr int([string index $Binary $i]*pow(2,$len-$i-1))]
    }
    return $sum
}
puts [Binary_Decimal 0011]
结果：3

十进制转换二进制

proc Decimal_Binary {Decimal} {
    set binary ""
    while {$Decimal!=0} {
        set binary "[expr $Decimal%2]$binary"
        set Decimal [expr $Decimal/2]
    }
    return $binary
}

puts [Decimal_Binary 8]
结果：1000

长度转换成子网掩码

proc Binary_Decimal {Binary} {
        set sum 0
        set len [string length $Binary]
        for {set i 0} {$i<$len} {incr i} {
            incr sum [expr int([string index $Binary $i]*pow(2,$len-$i-1))]
        }
        return $sum
    }

proc Len_Mask {len} {
    set mask ""
    for {set i 0} {$i<4} {incr i} {
        if {$len>=8} {
            lappend mask [Binary_Decimal [format 8 8]]
            incr len -8
        } else {
            lappend mask [Binary_Decimal [format $len 8]]
            set len 0
        }
    }
    return [join $mask .]
}

puts [Len_Mask 11]

proc Decimal_Binary {Decimal} {

proc Mask_Len {mask} {
    set msks [split $mask .]
    set len 0
    for {set i 0} {$i<4} {incr i} {
        set m [lindex $msks $i]
        set str [Decimal_Binary $m]
        incr len [string length [string trim $str 0]]
    }
    return $len
}

puts [Mask_Len 8.2.3.5]
